35 /** An [Iterator] to provide each generation from an initial [World]. The iterations will stop when the [World] has repeated itself.
36   * Note: On an unbounded board, with a glider, the iterations will continue ad infinitum.
37   * @param world
38   *   The initial [World].
39   */
40 final case class Generations(world: World) extends Iterator[World]:
41 
42   type WorldHistory = (World, List[World])
43 
44   private val initial: WorldHistory = (world, List.empty[World])
45 
46   private def nextWithHistory(current: WorldHistory): WorldHistory =
47     val next = current._1.nextGeneration
48     (next, next +: current._2)
49 
50   private val iterations = Iterator.iterate(initial)(nextWithHistory).buffered
51 
52   /** Peeks ahead to the next generation to determine if there is a `next`.
53     * @returns
54     *   true if the lookahead is a unique [World], false otherwise.
55     */
56   override def hasNext: Boolean =
57     val peek = iterations.head
58     peek._2.count(_ == peek._1) <= 1
59 
60   /** @returns
61     *   The next world. If the caller does not use `hasNext` then this will return the next generation continually, even if there
62     *   are no longer any living [Cell]s, or if history has repeated.
63     */
64   override def next: World = iterations.next._1
